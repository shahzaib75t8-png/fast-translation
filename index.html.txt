import React, { useState, useEffect, useRef } from 'react';
import { Languages, Copy, Check, ArrowLeftRight, Loader2, Eraser, AlertCircle, Sparkles, Volume2, Wand2, Info } from 'lucide-react';

const apiKey = ""; // API key is handled by the environment

const App = () => {
  const [inputText, setInputText] = useState("");
  const [translatedText, setTranslatedText] = useState("");
  const [mode, setMode] = useState("ru-to-en"); // Defaulting to Roman Urdu to English based on your usage
  const [isLoading, setIsLoading] = useState(false);
  const [isFixing, setIsFixing] = useState(false);
  const [isSpeaking, setIsSpeaking] = useState(false);
  const [copied, setCopied] = useState(false);
  const [error, setError] = useState(null);
  const [contextNote, setContextNote] = useState("");
  
  const audioRef = useRef(null);

  // Function to strip all symbols and emojis (Regex) per strict constraints
  const cleanText = (text) => {
    if (!text) return "";
    const symbolRegex = /[!@#$%^&*()"'`~;:,.?/\-_[\]{}|\\+=<>]/g;
    const emojiRegex = /[\u{1F600}-\u{1F64F}\u{1F300}-\u{1F5FF}\u{1F680}-\u{1F6FF}\u{1F1E0}-\u{1F1FF}\u{2600}-\u{26FF}\u{2700}-\u{27BF}]/gu;
    let cleaned = text.replace(symbolRegex, '').replace(emojiRegex, '');
    cleaned = cleaned.replace(/[^a-zA-Z0-9\s]/g, '');
    return cleaned.replace(/\s+/g, ' ').trim();
  };

  const callGemini = async (prompt, systemInstruction, model = "gemini-2.5-flash-preview-09-2025") => {
    let retries = 0;
    const maxRetries = 5;

    const execute = async () => {
      try {
        const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${apiKey}`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            contents: [{ parts: [{ text: prompt }] }],
            systemInstruction: { parts: [{ text: systemInstruction }] }
          })
        });

        if (!response.ok) throw new Error('API Error');
        const data = await response.json();
        return data.candidates?.[0]?.content?.parts?.[0]?.text || "";
      } catch (err) {
        if (retries < maxRetries) {
          retries++;
          await new Promise(res => setTimeout(res, Math.pow(2, retries) * 500));
          return execute();
        }
        throw err;
      }
    };
    return execute();
  };

  const handleTranslate = async () => {
    if (!inputText.trim()) return;
    setIsLoading(true);
    setError(null);
    setTranslatedText("");
    setContextNote("");

    try {
      // Improved prompts to prevent mirroring input
      const systemPrompt = mode === "en-to-ru" 
        ? "Translate English to Roman Urdu. DO NOT repeat the input. Output ONLY the translation. NO punctuation, symbols, or emojis."
        : "Translate Roman Urdu to English. DO NOT repeat the input. Output ONLY the translation. NO punctuation, symbols, or emojis.";

      const result = await callGemini(inputText, systemPrompt);
      
      // Check if AI just repeated the input
      if (result.toLowerCase().trim() === inputText.toLowerCase().trim()) {
        const fallbackResult = await callGemini(`Translate this specifically to ${mode === 'en-to-ru' ? 'Roman Urdu' : 'English'}: ${inputText}`, "Provide ONLY the translation. No symbols.");
        setTranslatedText(cleanText(fallbackResult));
      } else {
        setTranslatedText(cleanText(result));
      }

      const contextPrompt = `Analyze translation: "${inputText}" to "${result}". Tone and context in 5 words max.`;
      const contextResult = await callGemini(contextPrompt, "Short 5-word linguistic context.");
      setContextNote(contextResult);

    } catch (err) {
      setError("Translation failed. Try again.");
    } finally {
      setIsLoading(false);
    }
  };

  const handleFixGrammar = async () => {
    if (!inputText.trim()) return;
    setIsFixing(true);
    try {
      const prompt = `Fix grammar of this ${mode === "en-to-ru" ? "English" : "Roman Urdu"} text: "${inputText}". Return ONLY the fixed text.`;
      const fixed = await callGemini(prompt, "Professional editor. Return ONLY corrected text.");
      setInputText(fixed.trim());
    } catch (err) {
      setError("Couldn't polish text.");
    } finally {
      setIsFixing(false);
    }
  };

  const handleSpeak = async () => {
    if (!translatedText || isSpeaking) return;
    setIsSpeaking(true);
    try {
      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-tts:generateContent?key=${apiKey}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          contents: [{ parts: [{ text: `Say clearly: ${translatedText}` }] }],
          generationConfig: { 
            responseModalities: ["AUDIO"],
            speechConfig: { voiceConfig: { prebuiltVoiceConfig: { voiceName: "Kore" } } }
          }
        })
      });

      const data = await response.json();
      const pcmData = data.candidates?.[0]?.content?.parts?.[0]?.inlineData?.data;
      if (pcmData) {
        const audioBlob = new Blob([Uint8Array.from(atob(pcmData), c => c.charCodeAt(0))], { type: 'audio/l16' });
        const url = URL.createObjectURL(audioBlob);
        const audio = new Audio(url);
        audio.onended = () => setIsSpeaking(false);
        audio.play();
      }
    } catch (err) {
      setIsSpeaking(false);
    }
  };

  const handleCopy = () => {
    if (!translatedText) return;
    const textArea = document.createElement("textarea");
    textArea.value = translatedText;
    document.body.appendChild(textArea);
    textArea.select();
    document.execCommand('copy');
    document.body.removeChild(textArea);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };

  const toggleMode = () => {
    setMode(prev => prev === "en-to-ru" ? "ru-to-en" : "en-to-ru");
    setTranslatedText("");
    setContextNote("");
  };

  return (
    <div className="min-h-screen bg-slate-50 text-slate-900 font-sans p-4 md:p-8">
      <div className="max-w-5xl mx-auto">
        <header className="flex flex-col md:flex-row md:items-center justify-between gap-6 mb-10">
          <div className="flex items-center gap-4">
            <div className="bg-indigo-600 p-3 rounded-2xl text-white shadow-indigo-200 shadow-xl">
              <Languages size={32} />
            </div>
            <div>
              <h1 className="text-3xl font-black tracking-tighter text-slate-800 italic uppercase">Fast Translate</h1>
              <p className="text-[10px] font-bold text-slate-400 tracking-[0.2em] uppercase">AI Powered Logic</p>
            </div>
          </div>
          
          {/* Highlighted Toggle Button */}
          <button 
            onClick={toggleMode}
            className={`flex items-center justify-center gap-4 px-8 py-4 rounded-3xl transition-all shadow-lg active:scale-95 group font-black uppercase text-sm tracking-widest border-4 ${mode === 'ru-to-en' ? 'bg-indigo-600 text-white border-indigo-400' : 'bg-white text-indigo-600 border-indigo-100'}`}
          >
            <span className={mode === 'en-to-ru' ? 'underline decoration-2' : ''}>English</span>
            <ArrowLeftRight size={24} className="group-hover:rotate-180 transition-transform duration-500" />
            <span className={mode === 'ru-to-en' ? 'underline decoration-2' : ''}>Roman Urdu</span>
          </button>
        </header>

        <div className="grid grid-cols-1 md:grid-cols-2 gap-8">
          <div className="space-y-4">
            <div className="flex justify-between items-center px-2">
              <label className="text-[10px] font-black text-slate-400 uppercase tracking-[0.3em]">
                Input ({mode === 'en-to-ru' ? 'English' : 'Roman Urdu'})
              </label>
              <div className="flex gap-4">
                <button onClick={handleFixGrammar} disabled={isFixing || !inputText} className="text-[10px] font-black text-indigo-500 hover:text-indigo-700 flex items-center gap-1.5 uppercase">
                  {isFixing ? <Loader2 size={12} className="animate-spin" /> : <Wand2 size={12} />} Fix Polish
                </button>
                <button onClick={() => setInputText("")} className="text-[10px] font-black text-slate-400 hover:text-red-500 flex items-center gap-1.5 uppercase">
                  <Eraser size={12} /> Clear
                </button>
              </div>
            </div>
            <textarea
              className="w-full h-80 p-8 bg-white border-2 border-slate-100 rounded-[2.5rem] shadow-sm focus:ring-8 focus:ring-indigo-50/50 focus:border-indigo-400 outline-none resize-none transition-all text-xl font-medium"
              placeholder={mode === "en-to-ru" ? "Type English sentence..." : "Roman Urdu likhen..."}
              value={inputText}
              onChange={(e) => setInputText(e.target.value)}
            />
          </div>

          <div className="space-y-4">
            <div className="flex justify-between items-center px-2">
              <label className="text-[10px] font-black text-indigo-500 uppercase tracking-[0.3em]">
                Output ({mode === 'en-to-ru' ? 'Roman Urdu' : 'English'})
              </label>
              {contextNote && <span className="text-[10px] font-bold text-indigo-400 flex items-center gap-1 italic"><Info size={10} /> {contextNote}</span>}
            </div>
            <div className="relative group">
              <div className={`w-full h-80 p-8 rounded-[2.5rem] overflow-y-auto text-xl font-bold leading-relaxed border-2 transition-all ${translatedText ? 'bg-white border-indigo-100 shadow-xl text-slate-800' : 'bg-slate-50 border-slate-100 text-slate-300'}`}>
                {isLoading ? (
                  <div className="flex flex-col items-center justify-center h-full gap-4">
                    <Loader2 className="animate-spin text-indigo-600" size={40} />
                    <span className="text-xs font-black text-indigo-600 uppercase tracking-widest animate-pulse">Translating...</span>
                  </div>
                ) : (
                  translatedText || <span className="italic opacity-40 font-normal">Translation result...</span>
                )}
              </div>
              
              {translatedText && !isLoading && (
                <div className="absolute bottom-6 right-6 flex gap-3">
                  <button onClick={handleSpeak} disabled={isSpeaking} className="p-4 bg-slate-100 text-slate-600 rounded-2xl shadow-lg hover:bg-indigo-50 hover:text-indigo-600 transition-all active:scale-90">
                    {isSpeaking ? <Loader2 size={20} className="animate-spin" /> : <Volume2 size={20} />}
                  </button>
                  <button onClick={handleCopy} className="px-6 py-4 bg-indigo-600 text-white rounded-2xl shadow-xl hover:bg-indigo-700 transition-all flex items-center gap-2 active:scale-90">
                    {copied ? <Check size={20} /> : <Copy size={20} />}
                    <span className="text-[10px] font-black uppercase tracking-widest">{copied ? "Copied" : "Copy Result"}</span>
                  </button>
                </div>
              )}
            </div>
          </div>
        </div>

        <div className="mt-12 flex flex-col items-center">
          <button
            onClick={handleTranslate}
            disabled={isLoading || !inputText.trim()}
            className="group relative w-full md:w-96 overflow-hidden py-6 bg-slate-900 text-white rounded-[2rem] shadow-2xl hover:bg-black disabled:opacity-20 disabled:cursor-not-allowed transition-all transform active:scale-95"
          >
            <div className="relative z-10 flex items-center justify-center gap-3">
              <Sparkles className="text-yellow-400" size={20} />
              <span className="text-lg font-black uppercase tracking-widest">Translate Now</span>
            </div>
          </button>
        </div>
      </div>
    </div>
  );
};

export default App;
